# SMS Alarm Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add edge-triggered SMS alarm delivery to the existing collection loop while making `config.lua` the only configuration definition entrypoint.

**Architecture:** Keep the existing application loop and collection pipeline, then add a focused alarm evaluator and SMS wrapper above it. Move runtime configuration definitions into `config.lua`, keep `app_config.lua` as the persistence and validation layer, and wire `application.lua` so new alarms send one merged SMS without affecting normal collection or MQTT reporting.

**Tech Stack:** Lua, LuatOS `sms` and `fskv`, existing `application.lua` / `gmqtt.lua` / `app_collect.lua`, script-style Lua tests with fake globals, local Lua runtime at `D:\tool\lua\bin\lua.cmd`

```lua
local app_alarm = {}

function app_alarm.evaluate(cfg, snapshot, runtime, now_ms)
	local active_map = {}
	local previous_active = runtime and runtime.active_map or {}
	local door_open_since_ms = runtime and runtime.door_open_since_ms or nil
	local temp1 = snapshot.temp_hum and snapshot.temp_hum[0]
	local temp2 = snapshot.temp_hum and snapshot.temp_hum[1]
	local pressure1 = snapshot.pressure and snapshot.pressure[1]
	local pressure2 = snapshot.pressure and snapshot.pressure[2]

	if snapshot.door_open then
		door_open_since_ms = door_open_since_ms or now_ms
		if now_ms - door_open_since_ms >= cfg.door_open_warn_ms then
			active_map.door_open_timeout = true
		end
	else
		door_open_since_ms = nil
	end

	if temp1 and temp1.ok and temp1.temperature > cfg.temp_high then
		active_map.temp1_high = true
	end
	if snapshot.current_sensor_mv and snapshot.current_sensor_mv > cfg.current_high then
		active_map.current_high = true
	end
	if pressure1 and pressure1.ok and pressure2 and pressure2.ok and math.abs(pressure2.pressure - pressure1.pressure) < cfg.pressure_diff_low then
		active_map.pressure_diff_low = true
	end

	return {
		active_map = active_map,
		new_alarm_keys = new_alarm_keys,
		should_send_sms = #new_alarm_keys > 0,
		sms_text = sms_text,
		runtime = {
			active_map = active_map,
			door_open_since_ms = next_door_open_since_ms
		}
	}
end
```

local app_sms = {}

function app_sms.send_alert(phone, text)
	if type(phone) ~= "string" or phone == "" then
		return false
	end
	if type(text) ~= "string" or text == "" then
		return false
	end

	local action = sms.sendLong(phone, text, true)
	if type(action) ~= "table" or type(action.wait) ~= "function" then
		return false
	end

	return action.wait() and true or false
end
```

local app_alarm = rawget(_G, "app_alarm") or require("app_alarm")
local app_sms = rawget(_G, "app_sms") or require("app_sms")

local function now_ms()
	return (os.time() or 0) * 1000
end

local function start_collection_loop()
	sys.taskInit(function()
		local last_report_ms = 0
		local alarm_runtime = {
			active_map = {},
			door_open_since_ms = nil
		}

		while true do
			local cfg = app_config.get()
			local snapshot = app_collect.collect_once()
			local alarm = app_alarm.evaluate(cfg, snapshot, alarm_runtime, now_ms())

			alarm_runtime = alarm.runtime
			app_state.save_latest(snapshot)

			if alarm.should_send_sms then
				app_sms.send_alert(cfg.alarm_sms_phone, alarm.sms_text)
			end
		end
	end)
end
```
